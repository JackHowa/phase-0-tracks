CREATE TABLE day_types (
  id INTEGER PRIMARY KEY,
  name VARCHAR(255)
);

CREATE TABLE days (
  id INTEGER PRIMARY KEY,
  name VARCHAR(255),
  good BOOLEAN,
  day_type_id INT, 
  FOREIGN KEY (day_type_id) REFERENCES day_types(id)
);
